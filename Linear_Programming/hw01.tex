\begin{gather*}
    \max_{x_1, x_2, x_3}\quad Z = 5x_1 + 8x_2+6x_3\\
    \begin{aligned}
    \textup{s.t.}\quad  x_1 + x_2 + x_3 &\leq 12 \\
                        x_1 + 2x_2 + 2x_3 &\leq 20 \\
                        x_1 - 4x_2 + 3x_3 &= 15 \\
                        x_1, x_2, x_3 &\geq 0 \\
    \end{aligned}
\end{gather*}
